\section{Lesson 9}
\subsection{Memory Items}
\begin{enumerate}
    \item The Gram matrix of an $m\times n$ matrix is is an $n \times n$ (Hermitian) symmetric symmetric matrix that is always positive semi-definite (and often positive definite).
    \item The columns of a matrix are orthonormal if and only if the Gram matrix is the identity.
    \item The set of $n\times n$ orthogonal matrices $\text{O}(n)$ form an algebraic group:   i.e., (1) $\text{O}(n)$ is closed under multiplication, (2) multiplication is associative, (3) $I\in\text{O}(n)$, (4) each orthogonal matrix has a (multiplicative) inverse (easily computed as a transpose).
    \item The set of $2\times 2$ orthogonal matrices $\text{O}(2)$ consist of the rotation matrices (which have determinant equal to 1) and the reflection matrices (which have determinant equal to -1).
    \item The set of  permutation matrices $\text{S}_n$ forms an algebraic group (a subgroup of $\text{O}(n)$) that contains $n!$ elements.
    \item The action of an orthogonal matrix is an isometry.
\end{enumerate}
\subsection{Notes}
\begin{enumerate}
    \item Rotation and reflection matrices:
    \begin{align*}
        \text{Rot}(\theta) = \left[\begin{matrix}
            \cos\theta & -\sin\theta \\
            \sin\theta & \cos\theta \\
        \end{matrix}\right]
        \qquad
        \text{Ref}(\phi) = \left[\begin{matrix}
            \cos 2\phi & \sin 2\phi \\
            \sin 2\phi & -\cos 2\phi \\
        \end{matrix}\right]
    \end{align*}
\end{enumerate}
