\section{Lesson 18}

\subsection{Memory Items (Informal)}
\begin{enumerate}
    \item Scalar multiplication: $\alpha A = \alpha \, SDS^{-1}=S(\alpha D)S^{-1}$
    \item Shifts: $A+\alpha I ~=~SDS^{-1}+\alpha SIS^{-1}~=~S(D+\alpha I)S^{-1}$
    \item Powers: $A^p ~=~(SDS^{-1})^p~=~SD^pS^{-1}$
    \begin{enumerate}
        \item Applies to $A^-1$ as well!
    \end{enumerate}
    
\end{enumerate}
